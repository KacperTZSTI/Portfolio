#version 300 es
precision highp float;


in vec2 UV;
uniform vec2 mouse;
out vec4 out_color;
uniform float ratio, time;


const float PI = 3.14159264;


void main(void){
float x = UV.x * ratio;
float y = UV.y;
// Position of current point
vec2 p = vec2(x, y) - vec2(0.5 * ratio, 0.5);
vec3 color = vec3(0.0);
vec2 center = vec2(0.0);
float d = distance(p, center);


float phase = atan(p.x, p.y) + (d + time / PI) * 100.0;
float a = sin(phase);


float amp = smoothstep(0.45, 0.55, 0.5 + 0.5 * a);


color = mix(vec3(1.0,0.1,0.1), vec3(0.1,0.1,1.0), d * 1.42) * amp;
if (d < 0.015) {
color = vec3(1.0,0.0,0.0);
}


color = pow(color, vec3(0.4545));

out_color = vec4(color, 1.0);

}
