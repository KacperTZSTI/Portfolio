#version 300 es
precision highp float;


in vec2 UV;
uniform vec2 mouse;
out vec4 out_color;
uniform float ratio, time;


void main(void){
float x = UV.x * ratio;
float y = UV.y;


// Position of current point
vec2 p = vec2(x, y) - vec2(0.5 * ratio, 0.5);


vec4 col = vec4(0.0);


float t = 0.0;


t += (length(p)-0.1)/0.4;
t = 1.0-t*clamp(cos(t*30.0 + time * 6.2832),0.0,1.0) * (1.0-t);
t = 1.0 - t;
t *= 15.0;


col.r += t * cos(p.y * 60.0+ time * 6.2832 + 0.4);
col.g += t * cos(p.y * 61.1 + time * 6.2832 + 0.6);
col.b += t * cos(p.y * 63.0 + time * 6.2832 + 0.9);


float s = 0.0;
col.r += s;


col.b += 0.2;
col.rgb += 0.25 * cos(p.x * 1.0);


col.a = 1.0;


out_color = col;
}
