#version 300 es
precision highp float;


in vec2 UV;
uniform vec2 mouse;
out vec4 out_color;
uniform float ratio, time;


void main(void){
float x = UV.x * ratio;
float y = UV.y;


// Position of current point
vec2 p = vec2(x, y) - vec2(0.5 * ratio, 0.5);
p.x -= 0.2 * cos(p.y);
vec4 col = vec4(0.0);


vec2 center = vec2(0.0);


float f = mod(1.0 - p.x - time * 0.3, 0.3) * pow((p.x + 1.0), 2.0) * 1.6;



col.rgb += f;
col *= 1.0 - pow(length(p), 5.0);
float quant = 3.0;
float quant_offset_freq = 2000.0;
float quant_offset = cos(p.x * quant_offset_freq) * cos(p.y * quant_offset_freq);
col = floor(col * quant + quant_offset) / quant;


col.a = 1.0;


out_color = col;

}
